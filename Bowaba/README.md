# Partner Dashboard (Bowaba)

## Supabase env setup

1. Copy env template:

```bash
cp .env.example .env
```

2. Fill in your project values in `.env`:

```env
VITE_SUPABASE_URL=https://your-project-id.supabase.co
VITE_SUPABASE_ANON_KEY=your-anon-key
VITE_CLIENTS_TABLE=client_accounts
```

3. Restart dev server after changing `.env`.

## Required table schema

Run this in Supabase SQL Editor:

```sql
create table if not exists public.client_accounts (
  id bigint generated by default as identity primary key,
  client_id text not null unique check (client_id ~ '^[0-9]{6}$'),
  bm_id text,
  bm_name text,
  current_credit_balance numeric(12,2) default 0,
  current_automation_triggers_quota_left integer default 0,
  created_at timestamp with time zone default now()
);

-- Each client can have multiple WABA + phone number pairs
create table if not exists public.client_wabas (
  id bigint generated by default as identity primary key,
  client_id text not null references public.client_accounts(client_id) on delete cascade,
  waba_id text,
  phone_number text,
  created_at timestamp with time zone default now()
);

create index if not exists client_wabas_client_id_idx on public.client_wabas(client_id);
```

## Tightened RLS (recommended)

Use a mapping table so each authenticated user sees only allowed client records.

```sql
-- Enable RLS
alter table public.client_accounts enable row level security;

-- Mapping table: which user can access which client
create table if not exists public.user_client_access (
  user_id uuid not null references auth.users(id) on delete cascade,
  client_id text not null references public.client_accounts(client_id) on delete cascade,
  primary key (user_id, client_id)
);

alter table public.user_client_access enable row level security;

-- Users can read only their own mapping rows
drop policy if exists "read own client access" on public.user_client_access;
create policy "read own client access"
on public.user_client_access
for select
to authenticated
using (auth.uid() = user_id);

-- Users can read only client records mapped to them
drop policy if exists "read allowed clients only" on public.client_accounts;
create policy "read allowed clients only"
on public.client_accounts
for select
to authenticated
using (
  exists (
    select 1
    from public.user_client_access uca
    where uca.user_id = auth.uid()
      and uca.client_id = client_accounts.client_id
  )
);

-- Users can read WABA entries for their allowed clients
alter table public.client_wabas enable row level security;

drop policy if exists "read allowed client wabas" on public.client_wabas;
create policy "read allowed client wabas"
on public.client_wabas
for select
to authenticated
using (
  exists (
    select 1
    from public.user_client_access uca
    where uca.user_id = auth.uid()
      and uca.client_id = client_wabas.client_id
  )
);
```

Add access rows (example):

```sql
insert into public.user_client_access (user_id, client_id)
values ('<auth_user_uuid>', '329124');
```
